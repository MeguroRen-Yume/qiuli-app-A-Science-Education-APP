const jwtCheck = (req, res, next) => {
    const authHeader = req.headers['authorization'];
    if (typeof authHeader !== 'undefined') {
        const token = authHeader.split(' ')[1];
        jwt.verify(token, jwtSecret, (err, decoded) => {
            if (err) {
                console.error('JWT verification failed', err);
                res.status(403).json({ error: 'JWT verification failed' });
            } else {
                req.user = decoded; 
                next();
            }
        });
    } else {
        res.status(401).json({ error: 'Authorization header is missing' });
    }
};
module.exports = { jwtCheck };
const jwt = require('jsonwebtoken');

async function getUserId(token) {
    try {
        const decoded = jwt.verify(token, jwtSecret);
        if (isTokenExpired(decoded)) {
            console.log('Token has expired.');
            throw new Error('TokenExpiredError'); 
        }
        return decoded.userId;
    } catch (err) {
        console.error('Failed to verify token:', err);
        throw new Error('TokenInvalidError'); 
    }
}
function isTokenExpired(decodedToken) {
    if (!decodedToken.exp) return false; 
    const nowSeconds = Math.floor(Date.now() / 1000);
    return decodedToken.exp < nowSeconds;
}

function auth(req, res, next) {
  const token = req.header('x-auth-token');
  if (!token) {
    return res.status(401).json({ msg: 'No token, authorization denied' });
  }
  try {
    const decoded = jwt.verify(token, 'secretToken'); 
    req.user = decoded.user;
    next();
  } catch (err) {
    res.status(401).json({ msg: 'Token is not valid' });
  }
}

        return res.status(400).json({ success: false, message: '缺少必要的参数。' });
    }
    try {
        const userId = await getUserId(token);
        if (!userId) {
            return res.status(401).json({ success: false, message: '令牌无效或过期。' });
        }
        try {
            const insertGroupSql = 'INSERT INTO searcharticlehistory (userid, content) VALUES (?, ?)';
            const insertGroupParams = [userId, content];
            await db.execute(insertGroupSql, insertGroupParams);
            res.status(200).json({ success: true });
        } catch (insertError) {
            if (insertError.code === 'ER_DUP_ENTRY') {
                const updateSql = 'UPDATE searcharticlehistory SET created_at = CURRENT_TIMESTAMP WHERE userid = ? AND content = ?';
                await db.execute(updateSql, [userId, content]);
                res.status(200).json({ success: true });
            } else {
                console.error('Failed to insert or update record', insertError);
                res.status(500).json({ success: false });
            }
        }
    } catch (error) {
        console.error('Failed to handle request', error);
        res.status(500).json({ success: false });
    }

    if (!token || !content) {
        return res.status(400).json({ success: false, message: '缺少必要的参数。' });
    }
    try {
        const userId = await getUserId(token);
        if (!userId) {
            return res.status(401).json({ success: false, message: '令牌无效或过期。' });

        const [existingRecord] = await db.execute(
            'SELECT id FROM searchGroupHistory WHERE userid = ? AND content = ?',
            [userId, content]
        );
        if (existingRecord.length > 0) {
            const updateSql = 'UPDATE searchGroupHistory SET created_at = CURRENT_TIMESTAMP WHERE id = ?';
            await db.execute(updateSql, [existingRecord[0].id]);
        } else {
            const insertGroupSql = 'INSERT INTO searchGroupHistory (userid, content) VALUES (?, ?)';
            const insertGroupParams = [userId, content];
            await db.execute(insertGroupSql, insertGroupParams);
        }
        res.status(200).json({ success: true });
    } catch (error) {
        console.error('Failed to handle request', error);
        res.status(500).json({ success: false });
    }
